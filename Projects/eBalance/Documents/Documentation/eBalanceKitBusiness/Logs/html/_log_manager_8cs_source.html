00096         public void LogToDatabase(object msecs) {
00097             Dispose = false;
00098             int localMSec = 0;
00099             const int sleepTime = 100;
00100             while (!Dispose) {
00101                 Thread.Sleep(sleepTime);
00102                 localMSec += sleepTime;
00103                 if (Dispose) continue;
00104                 if (localMSec >= (int)msecs) {
00105                     if (LogActionBuffer.Count > 0) {
00106                         var actionLogs = new List<ActionLog>();
00107                         lock (LogActionBuffer) {
00108                             while (LogActionBuffer.Count > 0)
00109                                 actionLogs.Add(LogActionBuffer.Dequeue());
00110                             LogActionBuffer.Clear();
00111                         }
00112                         using (IDatabase conn = AppConfig.ConnectionManager.GetConnection()) {
00113                             foreach (ActionLog logAction in actionLogs)
00114                                 logAction.DoAction(conn);
00115                         }
00116                     }
00117                     if (LogBuffer.Count > 0) {
00118                         var logs = new List<LogEntryBase>();
00119                         lock (LogBuffer) {
00120                             while (LogBuffer.Count > 0)
00121                                 logs.Add(LogBuffer.Dequeue());
00122                             LogBuffer.Clear();
00123                         }
00124 
00125                         using (IDatabase conn = AppConfig.ConnectionManager.GetConnection()) {
00126                             try {
00127                                 conn.BeginTransaction();
00128                                 foreach (LogEntryBase log in logs)
00129                                     if (log != null)
00130                                         log.SaveToDb(conn);
00131                                 conn.CommitTransaction();
00132                             }
00133                             catch (Exception) {
00134                                 conn.RollbackTransaction();
00135                             }
00136                         }
00137                     }
00138                     localMSec = 0;
00139                 }
00140             }
00141         }

01381         internal void UpdateValue(AddValueInfo addValueInfo) {
01382             if (!Log) return;
01383             if (addValueInfo == null) return;

01385                 !(addValueInfo.Change == "AddToList" || addValueInfo.Change == "DeleteFromList" ||

01388                   addValueInfo.Change == "ValueOtherChanged"))

01391                 UpdateCompanyValue(addValueInfo);
01392             }
01393             else {
01394                 int taxonomyId = addValueInfo.Document != null
01395                                      ? addValueInfo.Document.TaxonomyIdManager.GetId(addValueInfo.TaxonomyString)
01396                                      : addValueInfo.Company.TaxonomyIdManager.GetId(addValueInfo.TaxonomyString);
01397 
01398                 var valueMapping = addValueInfo.NewValue as ValueMappingBase;

01400                 var valueChangeLog = new DbReportValueChangeLog {
01401                     NewValue = addValueInfo.NewValue.Value,

01403                     TaxonomyId = taxonomyId,
01404                     ReferenceId = valueMapping.Id,
01405                     UserId = GetUserId(),
01406                     TimeStamp = DateTime.Now
01407                 };
01408 
01409                 try {
01410                     valueChangeLog.ValueType = StringToValueType(addValueInfo.Change);

01416                     valueChangeLog.NewValue = addValueInfo.NewValue.ValueOther;
